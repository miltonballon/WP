CREATE TABLE `Factura` (
	`n_factura`	INTEGER,
	`n_autorizacion`	INTEGER,
	`id_proveedor`	INTEGER,
	`nit`	INTEGER,
	`fecha`	TEXT,
	FOREIGN KEY(`id_proveedor`) REFERENCES `Proveedor`(`id`),
	PRIMARY KEY(`n_factura`)
);
CREATE TABLE `Product` (
	`id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	`name`	TEXT
);
CREATE TABLE `Provider` (
	`id`	INTEGER PRIMARY KEY AUTOINCREMENT,
	`Provider`	TEXT
);
CREATE TABLE `Unit` (
	`id`	INTEGER PRIMARY KEY AUTOINCREMENT,
	`Type`	TEXT
);
CREATE TABLE `Product_Provider_Unit` (
	`Id_prod`	INTEGER,
	`id_prov`	INTEGER,
	`id_uni`	INTEGER,
	FOREIGN KEY(`id_prov`) REFERENCES `Provider`(`id`),
	FOREIGN KEY(`id_uni`) REFERENCES `Unit`(`id`),
	PRIMARY KEY(`Id_prod`,`id_prov`,`id_uni`),
	FOREIGN KEY(`Id_prod`) REFERENCES `Product`(`id`)
);
